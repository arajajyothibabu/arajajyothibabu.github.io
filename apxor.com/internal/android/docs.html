Apxor Android API

1. Initialization:

API: 

        
            ApxorSDK.initialize(String applicationID, String this.getApplicationContext());

Usage:

        
            ApxorSDK.initialize(“13f9df31-2367-444b-8dcf-8193783fd92e", this.getApplicationContext());

2. Setting UserId:

API: 

        
            ApxorSDK.setUserIdentifier(String userID);

Usage:

        
            ApxorSDK.setUserIdentifier(userID);

3. Setting User Details:

API:

        
            ApxorSDK.setUserCustomInfo(HashMap userCustomInfo);

Usage:

        
            HashMap userCustomInfo = new HashMap<>();
            userCustomInfo.put("location", "Hyderabad");
            userCustomInfo.put("country", “India");
            userCustomInfo.put("Source", “Twitter”);

            ApxorSDK.setUserCustomInfo(userCustomInfo);

4. Reporting Custom Error:

API:

        
            ApxorSDK.reportCustomError(String key);
            ApxorSDK.reportCustomError(String key, HashMap<String, String> errorInfo);
            ApxorSDK.reportCustomError(String key, HashMap<String, String> errorInfo, Exception exception);

Usage:

        
            Long number = null;
            try {
                number.toString();
            } catch (NullPointerException ex) {
                HashMap<String, String> additionalInfo = new HashMap<>();
                additionalInfo.put("cause", "tried to convert null value to string");
                additionalInfo.put("variable_name", "number");
                additionalInfo.put("variable_type", "Long");
                ApxorSDK.reportCustomError("Null Value", additionalInfo, ex);
            }

5. Set Navigation Item:

API:

        
            ApxorSDK.notifyCurrentVisibleViewDetails(String viewName, HashMap<String, String> additionalInfo);

Usage:

        
            HashMap<String, String> additionalInfo = new HashMap<>();

            additionalInfo.put("cardName", "sport");

            ApxorSDK.notifyCurrentVisibleViewDetails("NewsDetail", additionalInfo);

6. Log app event:

API:

        
            ApxorSDK.logAppEvent(String eventName);
            ApxorSDK.logAppEvent(String eventName, HashMap<String, String> additionalInfo);

Usage:

        
            HashMap<String, String> additionalInfo = new Hash
            additionalInfo.put("account", "private");
            additionalInfo.put("name", "jack");
            ApxorSDK.logAppEvent(“Logout”, additionalInfo);
